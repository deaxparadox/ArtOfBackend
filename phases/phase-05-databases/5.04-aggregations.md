## 1. Aggregate Functions

Aggregate functions collapse multiple rows into a single result:

```sql
COUNT(*)          -- count all rows (including NULLs)
COUNT(col)        -- count non-NULL values in col
COUNT(DISTINCT col)-- count unique non-NULL values
SUM(col)          -- sum of all non-NULL values
AVG(col)          -- average of non-NULL values
MIN(col)          -- smallest value
MAX(col)          -- largest value
BOOL_AND(col)     -- true if ALL values are true (PostgreSQL)
BOOL_OR(col)      -- true if ANY value is true (PostgreSQL)
STRING_AGG(col, sep) -- concatenate strings with separator (PostgreSQL)
ARRAY_AGG(col)    -- aggregate into array (PostgreSQL)
JSON_AGG(col)     -- aggregate into JSON array (PostgreSQL)
```

## 2. GROUP BY: Grouping Rows

```sql
-- Count orders per status
SELECT status, COUNT(*) AS count
FROM orders
GROUP BY status;

-- Revenue per day
SELECT
    DATE_TRUNC('day', created_at) AS date,
    COUNT(*)                      AS orders,
    SUM(total_cents) / 100.0      AS revenue_dollars
FROM orders
WHERE created_at >= NOW() - INTERVAL '30 days'
GROUP BY DATE_TRUNC('day', created_at)
ORDER BY date;

-- Top 10 customers by revenue
SELECT
    u.email,
    COUNT(o.id)           AS order_count,
    SUM(o.total_cents)    AS total_spent_cents
FROM users u
JOIN orders o ON u.id = o.user_id
WHERE o.status = 'paid'
GROUP BY u.id, u.email
ORDER BY total_spent_cents DESC
LIMIT 10;

-- HAVING: filter groups (must use after GROUP BY, unlike WHERE)
SELECT user_id, COUNT(*) AS order_count
FROM orders
GROUP BY user_id
HAVING COUNT(*) >= 5        -- only users with 5+ orders
   AND SUM(total_cents) > 10000;  -- who spent > $100

-- GROUP BY with ROLLUP (subtotals)
SELECT
    COALESCE(category, 'TOTAL') AS category,
    SUM(revenue_cents) AS revenue
FROM products
GROUP BY ROLLUP(category);
-- Returns: each category + a total row
```

## 3. Window Functions Deep Dive

Window functions compute values across rows without collapsing them. The `OVER()` clause defines the "window" (set of rows to consider for each calculation).

```sql
-- PARTITION BY: separate window per group
-- ORDER BY: order within the window
-- Frame: which rows in the window to include

-- Basic window function components:
function() OVER (
    PARTITION BY partition_column    -- reset window for each value
    ORDER BY sort_column             -- determines row order in window
    ROWS/RANGE BETWEEN ... AND ...   -- optional frame specification
)

-- ROW_NUMBER, RANK, DENSE_RANK: ranking
SELECT
    name,
    salary,
    department,
    ROW_NUMBER() OVER (PARTITION BY department ORDER BY salary DESC) AS rn,
    RANK()       OVER (PARTITION BY department ORDER BY salary DESC) AS rnk,
    DENSE_RANK() OVER (PARTITION BY department ORDER BY salary DESC) AS dense_rnk
FROM employees;
-- salary: 100, 90, 90, 80
-- ROW_NUMBER: 1, 2, 3, 4
-- RANK:       1, 2, 2, 4   (skips 3)
-- DENSE_RANK: 1, 2, 2, 3   (no skip)

-- Get highest-paid employee per department (top N per group)
SELECT name, salary, department
FROM (
    SELECT
        name, salary, department,
        ROW_NUMBER() OVER (PARTITION BY department ORDER BY salary DESC) AS rn
    FROM employees
) ranked
WHERE rn = 1;

-- Running total (cumulative sum)
SELECT
    order_date,
    daily_revenue,
    SUM(daily_revenue) OVER (
        ORDER BY order_date
        ROWS BETWEEN UNBOUNDED PRECEDING AND CURRENT ROW
    ) AS cumulative_revenue
FROM daily_revenue;

-- Moving average (rolling 7-day)
SELECT
    date,
    revenue,
    AVG(revenue) OVER (
        ORDER BY date
        ROWS BETWEEN 6 PRECEDING AND CURRENT ROW
    ) AS rolling_7day_avg
FROM daily_revenue;

-- LAG/LEAD: access previous/next rows
SELECT
    order_date,
    revenue,
    LAG(revenue, 1, 0) OVER (ORDER BY order_date) AS prev_day,  -- 0 as default
    LEAD(revenue, 1, 0) OVER (ORDER BY order_date) AS next_day,
    revenue - LAG(revenue) OVER (ORDER BY order_date) AS day_over_day_change
FROM daily_revenue;

-- FIRST_VALUE, LAST_VALUE, NTH_VALUE
SELECT
    name,
    salary,
    department,
    FIRST_VALUE(name) OVER (PARTITION BY department ORDER BY salary DESC) AS top_earner
FROM employees;

-- NTILE: divide into N buckets (percentiles)
SELECT
    name,
    salary,
    NTILE(4) OVER (ORDER BY salary) AS quartile  -- 1=bottom 25%, 4=top 25%
FROM employees;

-- PERCENT_RANK, CUME_DIST: relative ranking
SELECT
    name,
    salary,
    PERCENT_RANK() OVER (ORDER BY salary) AS percentile_rank,
    CUME_DIST() OVER (ORDER BY salary) AS cumulative_distribution
FROM employees;
```

## 4. Practical Analytics Patterns

```sql
-- Month-over-month growth
WITH monthly AS (
    SELECT
        DATE_TRUNC('month', created_at) AS month,
        SUM(total_cents) AS revenue
    FROM orders WHERE status = 'paid'
    GROUP BY 1
)
SELECT
    month,
    revenue,
    LAG(revenue) OVER (ORDER BY month) AS prev_revenue,
    ROUND(
        (revenue - LAG(revenue) OVER (ORDER BY month)) * 100.0
        / NULLIF(LAG(revenue) OVER (ORDER BY month), 0),
        2
    ) AS pct_growth
FROM monthly
ORDER BY month;

-- Cohort retention analysis
WITH cohorts AS (
    SELECT
        user_id,
        DATE_TRUNC('month', MIN(created_at)) AS cohort_month
    FROM orders GROUP BY user_id
),
activity AS (
    SELECT
        o.user_id,
        c.cohort_month,
        DATE_TRUNC('month', o.created_at) AS activity_month,
        EXTRACT(EPOCH FROM DATE_TRUNC('month', o.created_at) - c.cohort_month)
            / (30.5 * 86400) AS months_since_cohort
    FROM orders o JOIN cohorts c ON o.user_id = c.user_id
)
SELECT
    cohort_month,
    months_since_cohort,
    COUNT(DISTINCT user_id) AS active_users
FROM activity
GROUP BY 1, 2
ORDER BY 1, 2;
```

## 5. Senior Engineer Perspective

**The key insight about window functions:**

Window functions eliminate the most common reason to fetch data into application memory for processing. Instead of: fetch 10,000 rows into Python → sort → calculate running totals → aggregate — do it all in SQL. The database processes data 10-100x faster than moving it over the network and processing in Python.

**When to use GROUP BY vs window functions:**

- GROUP BY: reduce N rows to fewer rows (one row per group)
- Window functions: keep all N rows but add computed columns per row

If you need both: `GROUP BY` first, then window functions on the result.
