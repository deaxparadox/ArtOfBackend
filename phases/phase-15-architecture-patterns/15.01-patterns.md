## Domain-Driven Design (DDD) (15.1)

```python
# DDD: organize code around the business domain, not technical layers

# UBIQUITOUS LANGUAGE: shared vocabulary between devs and domain experts
# "Order" means the same thing in code, in meetings, in documentation

# BOUNDED CONTEXT: explicit boundary where a model applies
# "Order" in ordering context ≠ "Order" in fulfillment context

# AGGREGATE: cluster of domain objects treated as one unit
class Order:  # Aggregate root
    def __init__(self, order_id, user_id):
        self.id = order_id
        self.user_id = user_id
        self.items = []  # part of aggregate
        self.status = OrderStatus.PENDING
        self._events = []  # domain events

    def add_item(self, product_id, quantity, unit_price):
        if self.status != OrderStatus.PENDING:
            raise DomainException("Cannot add items to a non-pending order")
        item = OrderItem(product_id, quantity, unit_price)
        self.items.append(item)

    def place(self):
        if not self.items:
            raise DomainException("Cannot place an empty order")
        self.status = OrderStatus.PLACED
        self._events.append(OrderPlacedEvent(self.id, self.total_cents))

    @property
    def total_cents(self):
        return sum(item.subtotal for item in self.items)

# REPOSITORY: abstract data access
class OrderRepository:
    async def get_by_id(self, order_id: str) -> Order: ...
    async def save(self, order: Order) -> None: ...
    # No SQL here — that's an implementation detail

# VALUE OBJECT: immutable, identified by value not identity
from dataclasses import dataclass

@dataclass(frozen=True)  # immutable
class Money:
    amount_cents: int
    currency: str

    def __add__(self, other: "Money") -> "Money":
        if self.currency != other.currency:
            raise ValueError("Cannot add different currencies")
        return Money(self.amount_cents + other.amount_cents, self.currency)
```

## Hexagonal Architecture (Ports & Adapters) (15.2)

```python
# CORE: business logic has no knowledge of external systems
# PORTS: interfaces the core exposes or uses
# ADAPTERS: implementations of ports (database, HTTP, queue)

# Core / Domain layer (no imports from infrastructure)
class OrderService:
    def __init__(
        self,
        order_repo: OrderRepositoryPort,    # port (interface)
        payment_gateway: PaymentPort,        # port (interface)
        event_publisher: EventPublisherPort, # port (interface)
    ):
        self._orders = order_repo
        self._payments = payment_gateway
        self._events = event_publisher

    async def place_order(self, user_id: int, items: list[dict]) -> Order:
        order = Order.create(user_id, items)
        payment = await self._payments.charge(order.total_cents, user_id)
        order.mark_paid(payment.id)
        await self._orders.save(order)
        await self._events.publish(OrderPlacedEvent(order.id))
        return order

# Adapters (in infrastructure layer)
class PostgresOrderRepository(OrderRepositoryPort):
    async def save(self, order: Order):
        await db.execute("INSERT INTO orders ...", ...)

class StripePaymentGateway(PaymentPort):
    async def charge(self, amount_cents: int, user_id: int):
        return await stripe.charge(amount_cents, ...)

class KafkaEventPublisher(EventPublisherPort):
    async def publish(self, event):
        await kafka.produce(event.type, event.payload)

# Tests inject fake adapters — core logic is fully testable without real DB/Stripe/Kafka
class FakeOrderRepository(OrderRepositoryPort):
    def __init__(self): self.orders = {}
    async def save(self, order): self.orders[order.id] = order
```

## CQRS (15.6): Command Query Responsibility Segregation

```python
# CQRS: separate read and write models
# Writes: normalized, consistent (OLTP) — commands change state
# Reads: denormalized, optimized for queries — projections of state

# Write side: handles commands
class PlaceOrderCommand:
    user_id: int
    items: list[dict]

class OrderCommandHandler:
    async def handle(self, command: PlaceOrderCommand):
        order = Order.create(command.user_id, command.items)
        await order_repository.save(order)
        await event_bus.publish(OrderPlacedEvent(order.id))

# Read side: handles queries, reads from optimized projections
class OrderProjection:
    """Pre-computed read model — updated by event handlers."""
    # Stored in a separate table, optimized for common queries
    user_id: int
    order_count: int
    total_spent_cents: int
    last_order_date: datetime

@event_handler("order.placed")
async def update_order_projection(event: OrderPlacedEvent):
    """Maintain the read model when orders are placed."""
    await db.execute("""
        INSERT INTO order_projections (user_id, order_count, total_spent_cents, last_order_date)
        VALUES ($1, 1, $2, NOW())
        ON CONFLICT (user_id) DO UPDATE SET
            order_count = order_projections.order_count + 1,
            total_spent_cents = order_projections.total_spent_cents + $2,
            last_order_date = NOW()
    """, event.user_id, event.total_cents)

# Read query uses the projection (fast, no joins)
class OrderQueryHandler:
    async def get_user_summary(self, user_id: int) -> OrderProjection:
        return await db.fetchrow(
            "SELECT * FROM order_projections WHERE user_id = $1", user_id
        )
```

## Twelve-Factor App (15.12)

```
THE TWELVE FACTORS:

I.   CODEBASE:    One codebase tracked in VCS, many deploys
II.  DEPENDENCIES: Explicitly declare and isolate dependencies
                   (requirements.txt, package.json, go.mod)
III. CONFIG:       Store config in the environment, not code
IV.  BACKING SERVICES: Treat as attached resources (DB, queue, cache)
V.   BUILD/RELEASE/RUN: Strictly separate build and run stages
VI.  PROCESSES:    Stateless, share-nothing processes
VII. PORT BINDING: Self-contained service exports via port
VIII.CONCURRENCY:  Scale out via the process model
IX.  DISPOSABILITY: Fast startup, graceful shutdown
X.   DEV/PROD PARITY: Keep dev, staging, prod as similar as possible
XI.  LOGS:         Treat as event streams (stdout, not files)
XII. ADMIN PROCESSES: Run one-off admin tasks as one-off processes

MOST VIOLATED IN PRACTICE:
  III (config in env): credentials hardcoded in code
  VI (stateless):  session stored in application memory
  XI (logs to stdout): writing to files in container
  IX (graceful shutdown): not handling SIGTERM
```

## Database Sharding (15.15)

```python
# HORIZONTAL SHARDING: split table across multiple databases by range or hash

# HASH SHARDING: consistent distribution
def get_shard(user_id: int, num_shards: int = 4) -> int:
    return user_id % num_shards

# Route queries to correct shard
class ShardedUserRepository:
    def __init__(self, shard_connections: list):
        self.shards = shard_connections

    async def get_user(self, user_id: int) -> dict:
        shard = self.shards[user_id % len(self.shards)]
        return await shard.fetchrow(
            "SELECT * FROM users WHERE id = $1", user_id
        )

# CONSISTENT HASHING: add/remove shards without rehashing all data
# Nodes on a ring. Key maps to nearest clockwise node.
# Add node: only the new node's predecessor needs to migrate data.
# Much less data migration than modulo sharding.
```

## Consistent Hashing (15.16)

```python
import hashlib
import bisect

class ConsistentHashRing:
    def __init__(self, nodes: list[str], replicas: int = 150):
        self.replicas = replicas
        self.ring = {}
        self.sorted_keys = []

        for node in nodes:
            self.add_node(node)

    def add_node(self, node: str):
        for i in range(self.replicas):
            key = self._hash(f"{node}:{i}")
            self.ring[key] = node
            bisect.insort(self.sorted_keys, key)

    def remove_node(self, node: str):
        for i in range(self.replicas):
            key = self._hash(f"{node}:{i}")
            self.ring.pop(key, None)
            self.sorted_keys.remove(key)

    def get_node(self, key: str) -> str:
        hash_val = self._hash(key)
        idx = bisect.bisect(self.sorted_keys, hash_val) % len(self.sorted_keys)
        return self.ring[self.sorted_keys[idx]]

    def _hash(self, key: str) -> int:
        return int(hashlib.md5(key.encode()).hexdigest(), 16)

# Use: route cache keys to the correct Redis node
ring = ConsistentHashRing(["redis-1:6379", "redis-2:6379", "redis-3:6379"])
node = ring.get_node("user:123:profile")  # always routes to the same node
```

## Change Data Capture (CDC) (15.19)

```
CHANGE DATA CAPTURE: stream database changes as events
Use Debezium (most popular CDC tool) with Kafka:

PostgreSQL → Debezium (reads WAL/logical replication) → Kafka → Consumers

USE CASES:
  - Sync PostgreSQL data to Elasticsearch (search index)
  - Invalidate Redis cache when DB row changes
  - Replicate data to data warehouse for analytics
  - Audit trail of all database changes
  - Event sourcing from existing CRUD database

EXAMPLE FLOW:
  1. Enable logical replication in PostgreSQL
  2. Debezium connects as a replication client
  3. Every INSERT/UPDATE/DELETE in orders table → Kafka event
  4. Elasticsearch consumer: update product index
  5. Redis consumer: invalidate product cache
  6. Analytics consumer: insert into ClickHouse

DEBEZIUM EVENT FORMAT:
  {
    "op": "u",  // u=update, c=create, d=delete
    "before": {"id": 1, "status": "pending"},
    "after":  {"id": 1, "status": "paid"},
    "source": {"table": "orders", "ts_ms": 1718450200000}
  }
```
