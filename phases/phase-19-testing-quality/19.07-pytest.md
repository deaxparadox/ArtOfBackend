## 1. What Is pytest? (Theory)

pytest is the dominant Python testing framework. It replaces Python's built-in `unittest` with a simpler, more powerful approach: tests are just functions (no class inheritance required), fixtures provide dependency injection for test setup, and the plugin ecosystem handles everything from async support to code coverage.

**Why pytest over unittest:**

```python
# UNITTEST (verbose, OOP-heavy):
import unittest

class TestCalculator(unittest.TestCase):
    def setUp(self):
        self.calc = Calculator()
    
    def test_add(self):
        self.assertEqual(self.calc.add(2, 3), 5)
    
    def tearDown(self):
        pass

if __name__ == '__main__':
    unittest.main()

# PYTEST (clean, functional):
def test_add():
    calc = Calculator()
    assert calc.add(2, 3) == 5

# That's it. No class, no setUp/tearDown boilerplate.
# Pytest auto-discovers tests in any file starting with test_
```

## 2. Fixtures: pytest's Superpower

Fixtures provide shared setup code that can be injected into any test by name. They're much more powerful than setUp/tearDown.

```python
# conftest.py — fixtures available to all tests in the directory

import pytest
import asyncpg

@pytest.fixture(scope="session")
async def db_pool():
    """Database connection pool — created once per test session."""
    pool = await asyncpg.create_pool("postgresql://test:test@localhost/testdb")
    yield pool
    await pool.close()

@pytest.fixture(scope="function")  # default scope
async def clean_db(db_pool):
    """Clean database — runs before EACH test."""
    async with db_pool.acquire() as conn:
        await conn.execute("TRUNCATE users CASCADE")
    yield

@pytest.fixture
def fake_email():
    """Generate a unique email for test isolation."""
    import uuid
    return f"test-{uuid.uuid4().hex[:8]}@example.com"

@pytest.fixture
async def test_user(db_pool, fake_email):
    """Create a test user — available for tests that need one."""
    repo = UserRepository(db_pool)
    user = await repo.create(email=fake_email, name="Test User")
    yield user
    # cleanup (optional — clean_db handles this at test start)

# USING FIXTURES:
async def test_get_user(db_pool, clean_db, test_user):
    # db_pool: connection pool (session-scoped, created once)
    # clean_db: database truncated (function-scoped, runs each test)
    # test_user: User object created in database
    
    repo = UserRepository(db_pool)
    fetched = await repo.get_by_id(test_user.id)
    assert fetched.email == test_user.email
```

## 3. Fixture Scopes

```python
# SCOPE DETERMINES HOW LONG A FIXTURE LIVES:

@pytest.fixture(scope="session")
# Created once for the entire test session
# Teardown after all tests complete
# Use for: database connections, app servers

@pytest.fixture(scope="module")
# Created once per test module (file)
# Teardown after all tests in that module complete
# Use for: module-level expensive setup

@pytest.fixture(scope="function")   # DEFAULT
# Created fresh for each test
# Teardown after each test
# Use for: anything that needs to be clean per test

@pytest.fixture(scope="class")
# Created once per test class
# Teardown after class completes

# EXAMPLE: combining scopes efficiently
@pytest.fixture(scope="session")
def postgres_container():
    """Start one Docker container per test session (expensive)."""
    with PostgresContainer() as pg:
        yield pg  # container lives for entire session

@pytest.fixture(scope="session")
async def db_pool(postgres_container):
    """One pool per session (reuse connections)."""
    pool = await asyncpg.create_pool(postgres_container.url)
    yield pool
    await pool.close()

@pytest.fixture(autouse=True)   # runs for EVERY test automatically
async def clean_db(db_pool):
    """Clean state before EVERY test (function scope)."""
    await db_pool.execute("TRUNCATE users CASCADE")
    yield
```

## 4. Parametrize

```python
import pytest

# TEST MULTIPLE INPUTS WITHOUT DUPLICATING TEST CODE:
@pytest.mark.parametrize("input,expected", [
    ("hello", "HELLO"),
    ("world", "WORLD"),
    ("", ""),
    ("mixed Case", "MIXED CASE"),
])
def test_to_uppercase(input, expected):
    assert to_uppercase(input) == expected

# IDs: give each case a name for better test output
@pytest.mark.parametrize("email,valid", [
    ("alice@example.com", True),
    ("not-email", False),
], ids=["valid_email", "invalid_email"])
def test_email_validation(email, valid):
    assert is_valid_email(email) == valid

# INDIRECT PARAMETRIZE: pass parameter to a fixture
@pytest.fixture
def user_with_role(request, db_pool):
    role = request.param  # the parametrized value
    return create_user(db_pool, role=role)

@pytest.mark.parametrize("user_with_role,expected_access", [
    ("admin", True),
    ("viewer", False),
    ("editor", False),
], indirect=["user_with_role"])
def test_admin_only_access(user_with_role, expected_access):
    assert can_access_admin(user_with_role) == expected_access
```

## 5. Important Plugins

```python
# pytest-asyncio: async test support
# pip install pytest-asyncio

import pytest

@pytest.mark.asyncio
async def test_async_function():
    result = await some_async_function()
    assert result == expected

# In pytest.ini or pyproject.toml:
# [tool.pytest.ini_options]
# asyncio_mode = "auto"   # no need to mark every test

# ─────────────────────────────────────────────────────────
# pytest-cov: code coverage
# pip install pytest-cov
# $ pytest --cov=your_package --cov-report=html

# ─────────────────────────────────────────────────────────
# pytest-mock: simpler mocking syntax
# pip install pytest-mock

def test_with_mock(mocker):
    mock_service = mocker.patch("your_module.external_service")
    mock_service.call.return_value = {"status": "ok"}
    
    result = your_function_that_calls_service()
    
    mock_service.call.assert_called_once()
    assert result == "ok"

# ─────────────────────────────────────────────────────────
# factory-boy: test data factories
# pip install factory-boy

import factory

class UserFactory(factory.Factory):
    class Meta:
        model = User
    
    email = factory.Sequence(lambda n: f"user{n}@example.com")
    name = factory.Faker("name")
    created_at = factory.LazyFunction(datetime.utcnow)

def test_with_factory():
    user = UserFactory()         # User(email="user0@example.com", ...)
    users = UserFactory.batch(5) # 5 users with unique emails
```

## 6. pytest Configuration

```toml
# pyproject.toml

[tool.pytest.ini_options]
# Test discovery
testpaths = ["tests"]
python_files = ["test_*.py", "*_test.py"]
python_classes = ["Test*"]
python_functions = ["test_*"]

# Async mode
asyncio_mode = "auto"

# Warnings
filterwarnings = [
    "error",               # treat warnings as errors
    "ignore::DeprecationWarning:some_library",
]

# Markers (document custom markers)
markers = [
    "slow: marks tests as slow (deselect with '-m not slow')",
    "integration: marks tests as integration tests",
    "e2e: marks tests as end-to-end tests",
]

# Coverage
addopts = "--strict-markers --tb=short"
```

## 7. Senior Engineer Perspective

**Fixtures are the key to maintainable tests:**

Shared setup code in fixtures (not copy-pasted into each test) means: one change to the fixture updates all tests that use it. The alternative — setUp/tearDown copied into dozens of test files — creates a maintenance nightmare where you change the user creation logic in one place and break 40 tests in other files.

**The conftest.py hierarchy:**

pytest loads conftest.py files from the root down to the test directory. Fixtures in `tests/conftest.py` are available to all tests. Fixtures in `tests/integration/conftest.py` are only available to integration tests. Use this hierarchy to organize fixtures by scope and don't mix integration-specific fixtures with unit test fixtures.
