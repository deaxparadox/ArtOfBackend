## 1. Nginx: The Most Deployed Web Server (10.23)

Nginx (2004, Igor Sysoev) is simultaneously a web server, reverse proxy, load balancer, and HTTP cache. It powers approximately 34% of all websites globally, including Netflix, Dropbox, WordPress.com, and millions of others.

**Why Nginx is everywhere:**

Nginx uses an asynchronous, event-driven architecture — one worker process handles thousands of concurrent connections without the thread-per-connection overhead of Apache. Under high concurrency, Nginx uses dramatically less memory and CPU than traditional thread-based servers.

### The Production Nginx Configuration

```nginx
# /etc/nginx/nginx.conf

user www-data;
worker_processes auto;          # one worker per CPU core
worker_rlimit_nofile 65535;     # max file descriptors per worker

pid /run/nginx.pid;

events {
    worker_connections 4096;    # connections per worker
    multi_accept on;            # accept multiple connections per accept() call
    use epoll;                  # Linux: use epoll I/O event notification
}

http {
    # ─── BASICS ─────────────────────────────────────────────────
    include       /etc/nginx/mime.types;
    default_type  application/octet-stream;
    
    sendfile on;            # kernel-level file transfer (faster for static files)
    tcp_nopush on;          # send headers in one packet
    tcp_nodelay on;         # disable Nagle algorithm (lower latency)
    
    # ─── TIMEOUTS ────────────────────────────────────────────────
    keepalive_timeout 65s;          # keep-alive connection timeout
    keepalive_requests 1000;        # requests per keep-alive connection
    client_header_timeout 10s;      # how long to wait for request headers
    client_body_timeout 30s;        # how long to wait for request body
    send_timeout 30s;               # how long to wait for client to receive response
    
    # ─── PROXY TIMEOUTS ──────────────────────────────────────────
    proxy_connect_timeout 5s;       # timeout for connecting to backend
    proxy_send_timeout 30s;         # timeout for sending to backend
    proxy_read_timeout 30s;         # timeout for reading from backend
    
    # ─── BUFFER SIZES ────────────────────────────────────────────
    client_max_body_size 16M;       # max upload size (413 if exceeded)
    proxy_buffer_size 128k;
    proxy_buffers 4 256k;
    
    # ─── STRUCTURED LOGGING ──────────────────────────────────────
    log_format json_combined escape=json
        '{"time": "$time_iso8601", '
        '"method": "$request_method", '
        '"uri": "$request_uri", '
        '"status": $status, '
        '"bytes": $body_bytes_sent, '
        '"duration": "$request_time", '
        '"upstream_duration": "$upstream_response_time", '
        '"remote_addr": "$remote_addr", '
        '"user_agent": "$http_user_agent"}';
    
    access_log /var/log/nginx/access.log json_combined;
    error_log  /var/log/nginx/error.log warn;
    
    # ─── GZIP COMPRESSION ────────────────────────────────────────
    gzip on;
    gzip_comp_level 6;              # compression level 1-9 (6 is good balance)
    gzip_min_length 1000;           # don't compress tiny responses
    gzip_types
        text/plain text/css text/javascript
        application/json application/javascript
        application/x-javascript text/xml
        application/xml application/xml+rss;
    
    # ─── RATE LIMITING ───────────────────────────────────────────
    # Define a zone: 10MB shared memory, 100 requests/minute per IP
    limit_req_zone $binary_remote_addr zone=api_limit:10m rate=100r/m;
    
    # ─── UPSTREAM (BACKEND SERVERS) ──────────────────────────────
    upstream api_backend {
        least_conn;                  # route to backend with fewest active connections
        
        server 10.0.1.10:8000;
        server 10.0.1.11:8000;
        server 10.0.1.12:8000;
        
        keepalive 32;                # persistent connections to backends
    }
    
    # ─── HTTP → HTTPS REDIRECT ───────────────────────────────────
    server {
        listen 80;
        server_name api.example.com;
        return 301 https://$host$request_uri;
    }
    
    # ─── MAIN SERVER ─────────────────────────────────────────────
    server {
        listen 443 ssl http2;
        server_name api.example.com;
        
        # SSL
        ssl_certificate     /etc/letsencrypt/live/api.example.com/fullchain.pem;
        ssl_certificate_key /etc/letsencrypt/live/api.example.com/privkey.pem;
        ssl_protocols       TLSv1.2 TLSv1.3;
        ssl_ciphers         ECDHE-RSA-AES128-GCM-SHA256:ECDHE-RSA-AES256-GCM-SHA384;
        ssl_session_cache   shared:SSL:10m;
        ssl_session_timeout 1d;
        ssl_stapling        on;
        
        # Security headers
        add_header Strict-Transport-Security "max-age=31536000; includeSubDomains" always;
        add_header X-Content-Type-Options nosniff always;
        add_header X-Frame-Options DENY always;
        
        # API endpoints
        location /api/ {
            limit_req zone=api_limit burst=20 nodelay;  # allow bursts up to 20
            
            proxy_pass http://api_backend;
            proxy_http_version 1.1;
            proxy_set_header Connection "";          # enable keep-alive to upstream
            proxy_set_header Host $host;
            proxy_set_header X-Real-IP $remote_addr;
            proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
            proxy_set_header X-Forwarded-Proto https;
            proxy_set_header X-Request-ID $request_id;  # inject a request ID
        }
        
        # Health check endpoint (no rate limiting)
        location /health {
            proxy_pass http://api_backend;
            access_log off;  # don't log health checks (too noisy)
        }
        
        # Static files (served directly by Nginx, not proxied to backend)
        location /static/ {
            alias /var/www/static/;
            expires 1y;
            add_header Cache-Control "public, immutable";
            access_log off;
        }
    }
}
```

## 2. Caddy: Automatic HTTPS (10.24)

Caddy (2015, Matt Holt) is a modern web server that automatically provisions and renews Let's Encrypt TLS certificates. Zero configuration for HTTPS.

```
# Caddyfile — automatically handles TLS, HTTP→HTTPS redirect, HSTS

api.example.com {
    # That's it for HTTPS! Caddy automatically:
    # 1. Obtains a Let's Encrypt certificate for api.example.com
    # 2. Renews it before it expires
    # 3. Redirects HTTP to HTTPS
    # 4. Sets HSTS header
    
    reverse_proxy /api/* {
        to 10.0.1.10:8000 10.0.1.11:8000 10.0.1.12:8000
        lb_policy least_conn
        health_uri /health
        header_up X-Real-IP {remote_host}
    }
    
    # Rate limiting (via caddy-ratelimit plugin)
    rate_limit {
        zone api_limit {
            key {remote_host}
            events 100
            window 1m
        }
    }
    
    # Compression
    encode gzip
    
    # Security headers
    header {
        X-Content-Type-Options nosniff
        X-Frame-Options DENY
    }
    
    # Access log in JSON
    log {
        output file /var/log/caddy/access.log
        format json
    }
}
```

**Caddy vs. Nginx:**

```
CADDY:
  ✓ Automatic HTTPS: zero cert management overhead
  ✓ Simpler configuration syntax
  ✓ Modern HTTP/3 support by default
  ✗ Less ecosystem maturity than Nginx
  ✗ Fewer advanced features (rate limiting needs plugins)
  Best for: new projects, teams that want zero TLS ops burden

NGINX:
  ✓ Massive ecosystem, documentation, and community
  ✓ More advanced configuration options
  ✓ Battle-tested at enormous scale
  ✗ Manual TLS certificate management (or use certbot alongside)
  ✗ More complex configuration syntax
  Best for: teams that already know Nginx, complex routing requirements
```

## 3. Traefik: Cloud-Native Reverse Proxy (10.25)

Traefik (2015) auto-discovers services from Docker or Kubernetes and configures itself dynamically. You don't write Traefik config — Traefik reads it from your Docker labels or Kubernetes annotations.

```yaml
# With Traefik in Docker Compose:
services:
  traefik:
    image: traefik:v3.0
    command:
      - "--providers.docker=true"           # auto-discover from Docker
      - "--providers.docker.exposedbydefault=false"
      - "--entrypoints.web.address=:80"
      - "--entrypoints.websecure.address=:443"
      - "--certificatesresolvers.myresolver.acme.httpchallenge=true"
      - "--certificatesresolvers.myresolver.acme.email=admin@example.com"
    ports:
      - "80:80"
      - "443:443"
    volumes:
      - /var/run/docker.sock:/var/run/docker.sock

  api:
    image: myapp:latest
    labels:
      - "traefik.enable=true"
      - "traefik.http.routers.api.rule=Host(`api.example.com`)"
      - "traefik.http.routers.api.tls.certresolver=myresolver"
      - "traefik.http.services.api.loadbalancer.server.port=8000"
```

**Traefik is best for:** Docker and Kubernetes environments where services are added/removed frequently. Traefik watches for new containers and automatically routes traffic to them. No Nginx config reload needed.

## 4. HAProxy: High-Performance Load Balancing (10.26)

HAProxy (2000) is the standard for high-performance TCP and HTTP load balancing. It's used by GitHub, Reddit, Stack Overflow, and major CDNs for its exceptional performance and reliability.

```
# haproxy.cfg (simplified)
frontend api_frontend
    bind *:443 ssl crt /etc/ssl/myapp.pem
    mode http
    default_backend api_backend
    
    # Health check endpoint: respond directly without hitting backend
    acl health_check path_beg /health
    use_backend health_backend if health_check

backend api_backend
    balance leastconn
    option httpchk GET /health
    server api1 10.0.1.10:8000 check inter 5s rise 2 fall 3
    server api2 10.0.1.11:8000 check inter 5s rise 2 fall 3
    server api3 10.0.1.12:8000 check inter 5s rise 2 fall 3

# HAProxy stats page (password-protected)
frontend stats
    bind *:8404
    stats enable
    stats uri /stats
    stats auth admin:password
```

**HAProxy vs. Nginx for load balancing:**

HAProxy is purely a load balancer/proxy — it has no web server capabilities. Nginx can serve static files AND proxy. For pure load balancing workloads (especially TCP/L4), HAProxy often outperforms Nginx. For most web applications, either works fine.

## 5. Process Managers (10.27)

Process managers ensure your application starts on boot and restarts on crash.

### systemd (Linux standard)

```ini
# /etc/systemd/system/myapp.service
[Unit]
Description=My API Service
After=network.target

[Service]
Type=exec
User=appuser
Group=appuser
WorkingDirectory=/opt/myapp
ExecStart=/opt/myapp/venv/bin/uvicorn main:app --host 0.0.0.0 --port 8000 --workers 4
ExecReload=/bin/kill -HUP $MAINPID
Restart=on-failure
RestartSec=5s
StandardOutput=journal
StandardError=journal
SyslogIdentifier=myapp

# Resource limits
MemoryMax=2G
CPUQuota=200%

# Environment variables
EnvironmentFile=/etc/myapp/environment

[Install]
WantedBy=multi-user.target
```

```bash
# Enable and start
sudo systemctl enable myapp.service
sudo systemctl start myapp.service

# Check status
sudo systemctl status myapp.service

# View logs
sudo journalctl -u myapp.service -f  # follow
sudo journalctl -u myapp.service --since "1 hour ago"
```

### Supervisor (Python)

```ini
# /etc/supervisor/conf.d/myapp.conf
[program:myapp]
command=/opt/myapp/venv/bin/uvicorn main:app --host 0.0.0.0 --port 8000
directory=/opt/myapp
user=appuser
autostart=true
autorestart=true
startretries=3
stderr_logfile=/var/log/myapp/err.log
stdout_logfile=/var/log/myapp/out.log
environment=
    DATABASE_URL="postgresql://...",
    REDIS_URL="redis://..."

[program:worker]
command=/opt/myapp/venv/bin/celery -A app worker --loglevel=info
directory=/opt/myapp
user=appuser
numprocs=4  # start 4 worker processes
process_name=%(program_name)s_%(process_num)02d
```

### PM2 (Node.js)

```bash
# Start with PM2
pm2 start server.js --name "api" --instances 4 --watch

# Save configuration (survives reboots)
pm2 save
pm2 startup  # generates systemd/init script

# Status
pm2 status
pm2 logs api

# Reload without downtime (sends SIGINT, waits for graceful shutdown)
pm2 reload api
```
